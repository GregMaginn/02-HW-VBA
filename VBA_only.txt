Sub stockInfo():    Dim stockname As String    Dim stockvol As Double    Dim stockopen As Double    Dim stockclose As Double    Dim counter As Integer    Dim lastrow As Double        'defining variables    lastrow = Cells(Rows.Count, 1).End(xlUp).Row    stockname = Cells(2, 1)    stockopen = Cells(2, 3)    'set up table    Cells(1, 9).Value = "Ticker"    Cells(1, 10).Value = "Yearly Change"    Cells(1, 11).Value = " Percent Change"    Cells(1, 12).Value = "Total Stock Volume"        'variables for table    rowcounter = 2            For i = 2 To lastrow        If Cells(i, 1) <> stockname Then             'first do calculations on old stock and puit in table             Cells(rowcounter, 9) = stockname             Cells(rowcounter, 10) = stockclose - stockopen                          If stockopen = 0 Then                Cells(rowcounter, 11) = 0                             ElseIf stockopen <> 0 Then                Cells(rowcounter, 11) = (stockclose - stockopen) / (stockopen)             End If                          Cells(rowcounter, 12) = stockvol                          'add one to counter in total row            rowcounter = rowcounter + 1                        'set variables for new stock            stockname = Cells(i, 1)            stockopen = Cells(i, 3)            stockvol = Cells(i, 7)            'sets stockclose in case there is only one row for stock at i            stockclose = Cells(i, 6)                               'if next row is the same stock        ElseIf Cells(i, 1) = stockname Then          stockvol = stockvol + Cells(i, 7)          'set stockclose every time so last time will be correct          stockclose = Cells(i, 6)        End If            Next i        '**bonus    'only needs to run through new column to pick the highlights        'new variables    Dim highest As Double    Dim lowest As Double    Dim mostvol As Double        'new table    Cells(1, 15).Value = "Ticker"    Cells(1, 16).Value = "Value"        Cells(2, 14).Value = "Greatest % Increase"    Cells(3, 14).Value = "Greatest % Decrease"    Cells(4, 14).Value = "Greatest Volume"    'highest is row 2    'lowest is row 3    'most vol is row 4        'set first value to all and compare new rows against it    highest = Cells(2, 11)    lowest = Cells(2, 11)    mostvol = Cells(2, 12)        'tickert name for...    'increase    Cells(2, 15).Value = Cells(2, 9).Value    'decrease    Cells(3, 15).Value = Cells(2, 9).Value    'most volume    Cells(4, 15).Value = Cells(2, 9).Value        'increase    Cells(2, 16).Value = Cells(2, 11).Value    'decrease    Cells(3, 16).Value = Cells(2, 11).Value    'most volume    Cells(4, 16).Value = Cells(2, 12).Value        'rowcounter is already a variable that has a count of new rows    'run through all new rows    For i = 2 To rowcounter        If Cells(i, 11) > highest Then            highest = Cells(i, 11)            'set ticker            Cells(2, 15).Value = Cells(i, 9)            'set value            Cells(2, 16).Value = Cells(i, 11)        ElseIf Cells(i, 11) < lowest Then            lowest = Cells(i, 11)            'set ticker            Cells(3, 15).Value = Cells(i, 9)            'set value            Cells(3, 16).Value = Cells(i, 11)        End If        If Cells(i, 12) > mostvol Then            mostvol = Cells(i, 12)            'set ticker            Cells(4, 15).Value = Cells(i, 9)            'set value            Cells(4, 16).Value = Cells(i, 12)        End If                Next i            End Sub
